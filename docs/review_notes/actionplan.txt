# Setup dedicated logger for the Agent Runtime
logger = logging.getLogger("lorevox.chat_ws")
router = APIRouter(prefix="/api/chat", tags=["agent-runtime"])

# Load model once at module level to share VRAM across all WS connections
model, tokenizer = _load_model()

async def _ws_send(ws: WebSocket, payload: Dict[str, Any]) -> None:
    """Helper for safe JSON delivery over WebSocket."""
    try:
        await ws.send_text(json.dumps(payload, ensure_ascii=False))
    except Exception as e:
        logger.error(f"WS send failed: {e}")

async def _token_generator(
    conv_id: str, 
    user_message: str, 
    params: Dict[str, Any]
) -> AsyncIterator[str]:
    """
    Stateful token stream using TextIteratorStreamer.
    Offloads heavy GPU generation to a background thread to keep WS loop responsive.
    """
    # 1. Dynamically build the prompt (includes Memory + RAG in Phase 4)
    prompt = _build_prompt(conv_id, user_message)
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
    
    # 2. Setup the streamer
    streamer = TextIteratorStreamer(tokenizer, skip_prompt=True, skip_special_tokens=True)
    
    # 3. Define generation parameters (Blackwell-optimized)
    generation_kwargs = dict(
        **inputs,
        streamer=streamer,
        max_new_tokens=params.get("max_new_tokens", 2048),
        temperature=params.get("temperature", 0.7),
        top_p=params.get("top_p", 0.9),
        do_sample=True,
        pad_token_id=tokenizer.eos_token_id,
    )

    # 4. Run generation in background thread
    thread = Thread(target=model.generate, kwargs=generation_kwargs)
    thread.start()

    # 5. Yield tokens as they arrive
    for token in streamer:
        yield token
        await asyncio.sleep(0)  # Yield control to the event loop

@router.websocket("/ws")
async def agent_websocket_endpoint(websocket: WebSocket):
    """
    Phase 1 WebSocket Endpoint: ws://localhost:8000/api/chat/ws
    Handles bidirectional 'start_turn' and 'cancel_turn' protocol.
    """
    await websocket.accept()
    logger.info("Agent WS connection accepted.")

    current_task: Optional[asyncio.Task] = None
    
    try:
        while True:
            # Receive structured JSON frame from UI
            data = await websocket.receive_json()
            msg_type = data.get("type")
            
            if msg_type == "start_turn":
                # Ensure only one turn is active at a time (Phase 1 rule)
                if current_task and not current_task.done():
                    current_task.cancel()
                
                conv_id = data.get("session_id", "default")
                user_msg = data.get("message", "")
                params = data.get("params", {})
                turn_id = params.get("turn_id", f"turn-{int(asyncio.get_event_loop().time())}")

                async def run_turn(t_id: str):
                    full_response = ""
                    try:
                        await _ws_send(websocket, {"type": "status", "turn_id": t_id, "state": "thinking"})
                        
                        async for token in _token_generator(conv_id, user_msg, params):
                            full_response += token
                            await _ws_send(websocket, {"type": "token", "turn_id": t_id, "delta": token})
                        
                        # Phase 3: Post-turn DB write after streaming finishes
                        persist_turn_transaction(
                            conv_id=conv_id,
                            user_message=user_msg,
                            assistant_message=full_response,
                            model_name=model.name_or_path
                        )
                        
                        await _ws_send(websocket, {"type": "done", "turn_id": t_id, "final_text": full_response})
                        
                    except asyncio.CancelledError:
                        logger.info(f"Turn {t_id} cancelled.")
                    except Exception as e:
                        logger.error(f"Error in turn {t_id}: {e}")
                        await _ws_send(websocket, {"type": "error", "turn_id": t_id, "message": str(e)})

                current_task = asyncio.create_task(run_turn(turn_id))

            elif msg_type == "cancel_turn":
                if current_task and not current_task.done():
                    current_task.cancel()
                    await _ws_send(websocket, {"type": "status", "state": "cancelled"})

            elif msg_type == "ping":
                await _ws_send(websocket, {"type": "pong"})

    except WebSocketDisconnect:
        logger.info("Agent WS disconnected.")
        if current_task:
            current_task.cancel()
    finally:
        await websocket.close()

/* Phase 3: WebSocket + Fallback Config */
const WS_URL = ORIGIN.replace("http", "ws") + "/api/chat/ws";
let ws = null;
let useFallback = false;

function connectAgent() {
  ws = new WebSocket(WS_URL);

  ws.onopen = () => {
    useFallback = false;
    on("pillChat");
    systemBubble("Connected to Agent Runtime (WS)");
  };

  ws.onclose = () => {
    useFallback = true;
    err("pillChat");
    console.warn("WS Disconnected. Falling back to SSE mode.");
    setTimeout(connectAgent, 5000); // Reconnect attempt
  };

  ws.onmessage = (event) => {
    const msg = JSON.parse(event.data);
    handleAgentMessage(msg);
  };
}

/* Phase 3: Unified Send Logic */

      type: "start_turn",

      message: text,
      params: { temperature: 0.7 }

    await streamChat(text);

@router.post("/transcribe")
async def stt_transcribe(file: UploadFile = File(...)):
    kind, engine = _load_stt_engine()
    if not engine: raise HTTPException(501, "STT engine offline.")
    
    tmpdir = pathlib.Path(tempfile.mkdtemp(prefix="stt_"))
    try:
        audio_path = tmpdir / "upload.webm"
        with open(audio_path, "wb") as f: f.write(await file.read())
        
        segments, _ = engine.transcribe(str(audio_path), vad_filter=True)
        text = " ".join(s.text for s in segments).strip()
        return {"ok": True, "text": text}
    finally: shutil.rmtree(tmpdir, ignore_errors=True)

def assemble_memoir_context(conv_id: str):
    with connect_db() as con:
        # Pull facts, timeline events, and narrative voice
        profile = con.execute("SELECT profile_json FROM profiles p JOIN sessions s ON p.person_id=s.person_id WHERE s.session_id=?", (conv_id,)).fetchone()
        timeline = con.execute("SELECT event_date, event_title FROM timeline_events e JOIN sessions s ON e.person_id=s.person_id WHERE s.session_id=?", (conv_id,)).fetchall()
        transcript = con.execute("SELECT role, content FROM turns WHERE conv_id=? ORDER BY ts ASC", (conv_id,)).fetchall()
    
    return {
        "profile": profile[0] if profile else "{}",
        "timeline": "\n".join([f"- {r[0]}: {r[1]}" for r in timeline]),
        "voice": "\n".join([f"{r[0].upper()}: {r[1]}" for r in transcript])
    }

@router.post("/chapter")
async def draft_chapter(conv_id: str):
    context = assemble_memoir_context(conv_id)
    # Master prompt using Mark Twain / Joan Didion instructions
    prompt = f"### DATA\nProfile: {context['profile']}\nTimeline: {context['timeline']}\nVoice: {context['voice']}\n\nWrite Chapter 1: Beginnings."

    path = pathlib.Path(f"/home/chris/lorevox_data/exports/chapter_{conv_id}.md")
