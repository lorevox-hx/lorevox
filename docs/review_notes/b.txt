async def summarize_history(session_id: str, history: list[dict[str, str]]) -> str:
    text = ""
    for msg in history[-8:]:
        role = msg["role"]
        content = msg["content"]
        text += f"{role.upper()}: {content}\n"

    prompt = (
        "Summarize the key stable facts, preferences, and biography details "
        "from this conversation so far. Be concise.\n\n" + text
    )

    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
    out = model.generate(**inputs, max_new_tokens=256)
    summary = tokenizer.decode(
        out[0][inputs["input_ids"].shape[1]:],
        skip_special_tokens=True,
    )
    return summary.strip()

async def run_tool(session_id: str, tool_name: str, arguments: dict) -> dict | str:
    """
    Executes real-world actions triggered by the LLM.
    Tools:
      - update_profile
      - add_family_member
      - extract_timeline_event
      - prompt_photo_upload
    """
    conn = get_conn()
    try:
        # Map session -> person_id if sessions table exists
        person_id = None
        try:
            row = conn.execute(
                "SELECT person_id FROM sessions WHERE session_id = ?",
                (session_id,),
            ).fetchone()
            if row:
                person_id = row[0]
        except Exception:
            # sessions table may not exist yet; tools that need person_id will fail gracefully
            person_id = None

        # 1) Profile Editor
        if tool_name == "update_profile":
            path = arguments.get("path")
            value = arguments.get("value")
            if not path or value is None:
                return {"error": "Missing path or value"}

            if not person_id:
                return {"error": "No person_id for this session"}

            upsert_profile_field(person_id, path, value)
            return {"status": "success", "field": path, "value": value}

        # 2) Family Member
        if tool_name == "add_family_member":
            name = arguments.get("name")
            relation = arguments.get("relation")
            if not name or not relation:
                return {"error": "Missing name or relation"}

            if not person_id:
                return {"error": "No person_id for this session"}

            # create person + relationship
            conn.execute(
                "INSERT INTO people (display_name, role) VALUES (?, ?)",
                (name, relation.lower()),
            )
            new_id = conn.execute("SELECT last_insert_rowid()").fetchone()[0]
            conn.execute(
                "INSERT INTO relationships (person_id, related_person_id, relationship_type) "
                "VALUES (?, ?, ?)",
                (person_id, new_id, relation),
            )
            conn.commit()
            return {"status": "success", "added": name, "relation": relation}

        # 3) Timeline Event
        if tool_name == "extract_timeline_event":
            event_date = arguments.get("event_date")
            event_title = arguments.get("event_title")
            description = arguments.get("description")
            location = arguments.get("location")

            if not event_date or not event_title:
                return {"error": "Missing event date or title"}

            if not person_id:
                return {"error": "No person_id for this session"}

            event_id = add_timeline_event(
                person_id=person_id,
                event_date=event_date,
                event_title=event_title,
                event_description=description or "",
                location_name=location or "",
            )
            return {
                "status": "success",
                "event_id": event_id,
                "title": event_title,
            }

        # 4) Photo Attachment Prompt
        if tool_name == "prompt_photo_upload":
            context = arguments.get("context")
            event_id = arguments.get("event_id")
            return {
                "status": "pending_upload",
                "context": context,
                "event_id": event_id,
                "instruction": "Please select a photo to attach to this memory.",
            }

        return {"error": f"Unknown tool: {tool_name}"}

    finally:
        conn.close()

async def fetch_rag_context(session_id: str, message: str) -> str:
    q_emb = embed_text(message)
    chunks = fetch_top_k_chunks(q_emb, k=5)
    return "\n\n".join(chunks)


def build_messages(session_id: str, user_message: str, rag: str) -> list[dict[str, str]]:
    history = SESSION_HISTORY.get(session_id, [])
    memory = get_session_memory(session_id)

    system_prompt = (
        "You are Lori, a memoir/oral-history interviewer for LoreVox.\n"
        "Your goals:\n"
        "1) Ask one question at a time from the interview plan.\n"
        "2) Listen carefully and respond with empathy.\n"
        "3) When the user provides hard facts (names, dates, places, occupations), "
        "use tools instead of normal text.\n\n"
        "TOOLS:\n"
        "1. update_profile\n"
        "   Format: {\"tool_name\": \"update_profile\", \"arguments\": {\"path\": \"personal.full_name\", \"value\": \"Christopher Todd Horne\"}}\n"
        "2. add_family_member\n"
        "   Format: {\"tool_name\": \"add_family_member\", \"arguments\": {\"name\": \"Gretchen\", \"relation\": \"Daughter\"}}\n"
        "3. extract_timeline_event\n"
        "   Format: {\"tool_name\": \"extract_timeline_event\", \"arguments\": {\"event_date\": \"2026-05-01\", \"event_title\": \"Retirement\", \"description\": \"Left my job.\", \"location\": \"New Mexico\"}}\n"
        "4. prompt_photo_upload\n"
        "   Format: {\"tool_name\": \"prompt_photo_upload\", \"arguments\": {\"context\": \"Photo of childhood home\", \"event_id\": \"123\"}}\n\n"
        "IMPORTANT:\n"
        "- When you output a tool call, output ONLY the JSON object, no extra text.\n"
        "- For normal conversation, respond as Lori in natural language.\n\n"
        f"[MEMORY]\n{memory}\n\n[CONTEXT]\n{rag}\n"
    )

    msgs: list[dict[str, str]] = [{"role": "system", "content": system_prompt}]
    msgs.extend(history)
    msgs.append({"role": "user", "content": user_message})
    return msgs


# =========================
#  GENERATION
# =========================

async def generate_tokens(
    session_id: str,
    message: str,
    params: Optional[Dict[str, Any]] = None,
) -> AsyncIterator[str]:
    params = params or {}

    rag = await fetch_rag_context(session_id, message)
    messages = build_messages(session_id, message, rag)

    inputs = tokenizer.apply_chat_template(
        messages,
        return_tensors="pt",
    ).to(model.device)

    streamer = TextIteratorStreamer(
        tokenizer,
        skip_prompt=True,
        skip_special_tokens=True,
    )

    generation_kwargs = dict(
        **inputs,
        streamer=streamer,
        max_new_tokens=params.get("max_new_tokens", 2048),
        temperature=params.get("temperature", 0.7),
        top_p=params.get("top_p", 0.9),
    )

    thread = Thread(target=model.generate, kwargs=generation_kwargs)
    thread.start()

    for token in streamer:
        yield token
        await asyncio.sleep(0)

@router.websocket("/stream")
async def chat_stream(ws: WebSocket):
    await ws.accept()

    current_task: Optional[asyncio.Task] = None
    current_turn_id: Optional[str] = None

    try:
        while True:
            raw = await ws.receive_text()
            msg = json.loads(raw)
            msg_type = msg.get("type")

            # START TURN
            if msg_type == "start_turn":
                if current_task and not current_task.done():
                    current_task.cancel()

                session_id = msg.get("session_id", "default")
                user_message = msg["message"]
                params = msg.get("params") or {}
                model_name = params.get("model_name", MODEL_ID)
                tts_enabled = params.get("tts", False)

                current_turn_id = params.get("turn_id", f"turn-{int(asyncio.get_event_loop().time()*1000)}")

                await ws.send_text(json.dumps({
                    "type": "status",
                    "turn_id": current_turn_id,
                    "state": "thinking",
                }))

                async def run_turn():
                    full_text_parts: list[str] = []

                    try:
                        await ws.send_text(json.dumps({
                            "type": "status",
                            "turn_id": current_turn_id,
                            "state": "working",
                        }))

                        async for delta in generate_tokens(
                            session_id=session_id,
                            message=user_message,
                            params=params,
                        ):
                            full_text_parts.append(delta)
                            await ws.send_text(json.dumps({
                                "type": "token",
                                "turn_id": current_turn_id,
                                "delta": delta,
                            }))

                        raw_full_text = "".join(full_text_parts)

                        # TOOL CALL DETECTION
                        tool_result = None
                        if raw_full_text.strip().startswith("{") and '"tool_name"' in raw_full_text:
                            try:
                                tool_call = json.loads(raw_full_text)
                                tool_name = tool_call["tool_name"]
                                arguments = tool_call.get("arguments", {})

                                tool_result = await run_tool(session_id, tool_name, arguments)

                                log_tool_call(
                                    session_id=session_id,
                                    turn_id=current_turn_id,
                                    tool_name=tool_name,
                                    arguments=arguments,
                                    result=tool_result,
                                )

                                await ws.send_text(json.dumps({
                                    "type": "tool",
                                    "turn_id": current_turn_id,
                                    "tool_name": tool_name,
                                    "arguments": arguments,
                                    "result": tool_result,
                                }))

                                # Silent tool: don't show JSON to user
                                full_text = ""
                            except Exception as e:
                                print("Tool call error:", e)
                                full_text = ""
                        else:
                            full_text = raw_full_text

                        # UPDATE HISTORY
                        SESSION_HISTORY.setdefault(session_id, []).extend([
                            {"role": "user", "content": user_message},
                            {"role": "assistant", "content": full_text},
                        ])

                        # PERSIST TURN
                        write_chat_turn(
                            session_id=session_id,
                            user_message=user_message,
                            assistant_message=full_text,
                            model_name=model_name,
                            metadata={
                                "params": params,
                                "token_count": len(full_text),
                            },
                        )

                        # SNAPSHOT PROFILE (if you want per-turn diffs)
                        snapshot_profile(session_id, current_turn_id)

                        # UPDATE MEMORY
                        summary = await summarize_history(
                            session_id,
                            SESSION_HISTORY[session_id],
                        )
                        upsert_session_memory(session_id, summary)

                        # TTS
                        if tts_enabled and full_text:
                            t = asyncio.create_task(
                                tts_speak(current_turn_id, full_text)
                            )
                            TTS_TASKS[current_turn_id] = t

                        await ws.send_text(json.dumps({
                            "type": "done",
                            "turn_id": current_turn_id,
                            "final_text": full_text,
                        }))

                    except asyncio.CancelledError:
                        pass
                    except Exception as e:
                        await ws.send_text(json.dumps({
                            "type": "error",
                            "turn_id": current_turn_id,
                            "message": str(e),
                        }))

                current_task = asyncio.create_task(run_turn())

            # CANCEL TURN
            elif msg_type == "cancel_turn":
                turn_id = msg.get("turn_id")
                if current_task and not current_task.done():
                    current_task.cancel()
                    current_task = None

                tts_task = TTS_TASKS.get(turn_id)
                if tts_task and not tts_task.done():
                    tts_task.cancel()

                await ws.send_text(json.dumps({
                    "type": "status",
                    "turn_id": turn_id,
                    "state": "cancelled",
                }))

            # PING
            elif msg_type == "ping":
                await ws.send_text(json.dumps({"type": "pong"}))

            else:
                await ws.send_text(json.dumps({
                    "type": "error",
                    "message": f"Unknown message type: {msg_type}",
                }))

    except WebSocketDisconnect:
        if current_task and not current_task.done():
            current_task.cancel()
    except Exception as e:
        await ws.send_text(json.dumps({
            "type": "error",
            "message": str(e),
        }))
        if current_task and not current_task.done():
            current_task.cancel()
        await ws.close()
