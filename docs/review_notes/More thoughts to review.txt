def assemble_memoir_context(session_id: str) -> Dict[str, str]:
    """
    Fetches and formats all agentic data layers for a specific session
    to be used in the Memoir Draft prompt.
    """
    from .db import _connect  # Uses your existing WAL-enabled connection
    
    with _connect() as con:
        # 1. Fetch the Profile (Hard Facts)
        # Links session to the person and their biographical profile
        person_row = con.execute('''
            SELECT p.person_id, pr.profile_json 
            FROM sessions s
            JOIN people p ON s.person_id = p.person_id
            JOIN profiles pr ON p.person_id = pr.person_id
            WHERE s.session_id = ?
        ''', (session_id,)).fetchone()
        
        person_id = person_row['person_id']
        profile_data = person_row['profile_json']

        # 2. Fetch the Timeline (Chronology)
        # Gathers dated events for structural accuracy
        timeline_rows = con.execute('''
            SELECT event_date, event_title, event_description, location_name
            FROM timeline_events
            WHERE person_id = ?
            ORDER BY event_date ASC
        ''', (person_id,)).fetchall()
        
        timeline_txt = "\n".join([
            f"- {r['event_date']}: {r['event_title']} ({r['location_name']}) - {r['event_description']}"
            for r in timeline_rows
        ])

        # 3. Fetch Media Metadata (Sensory Details)
        # Provides visual anchors for the narrative
        media_rows = con.execute('''
            SELECT description, taken_at, location_name
            FROM media
            WHERE person_id = ?
        ''', (person_id,)).fetchall()
        
        media_txt = "\n".join([
            f"- Image from {r['taken_at']} at {r['location_name']}: {r['description']}"
            for r in media_rows
        ])

        # 4. Fetch Transcripts (The Voice) [cite: 14, 19]
        # Captures raw narrative and emotional subtext
        turn_rows = con.execute('''
            SELECT role, content FROM turns 
            WHERE conv_id = ? 
            ORDER BY ts ASC
        ''', (session_id,)).fetchall()
        
        transcript_txt = "\n".join([f"{r['role'].upper()}: {r['content']}" for r in turn_rows])

    return {
        "profile_json": profile_data,
        "timeline_events": timeline_txt or "No timeline events recorded.",
        "media_descriptions": media_txt or "No media descriptions available.",
        "chat_turns": transcript_txt
    }

2. Implement the Streaming Logic (JavaScript)
This function calls your backend, which assembles the Profile, Timeline, Media, and Transcripts. It then uses a WebSocket-like stream to show the text as it's being "written" by the model.
JavaScript
async function generateChapterDraft() {
  const content = document.getElementById("draftContent");
  content.innerHTML = '<div class="animate-pulse text-amber-500">Assembling life data and drafting...</div>';
  
  try {
    // This calls the assembly function we built for the backend
    const res = await fetch(`/api/draft/chapter?session_id=${state.conv_id}`, { method: "POST" });
    const reader = res.body.getReader();
    const decoder = new TextDecoder();
    
    content.innerHTML = ""; // Clear the loading state
    let fullDraft = "";

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      const chunk = decoder.decode(value, { stream: true });
      fullDraft += chunk;
      content.innerText = fullDraft;
      
      // Auto-scroll the draft panel as it grows
      const panel = document.getElementById("draftTab");
      panel.scrollTop = panel.scrollHeight;
    }
    
    setStatus("Chapter 1 Draft Complete.");
    systemBubble("?? Memoir chapter drafted successfully.");
  } catch (e) {
    content.innerHTML = '<span class="text-red-500">Drafting failed. Ensure the LLM service is active.</span>';
  }
}
3. Backend Integration Review
For this UI to work, your main.py must be configured to handle the assembly of these four critical layers:
* Profile: Used for hard-fact consistency.
* Timeline: Used to set the chronological structure of the chapter.
* Media: Used to provide descriptive, sensory anchors.
* Transcript: Used to preserve your unique narrative voice.
