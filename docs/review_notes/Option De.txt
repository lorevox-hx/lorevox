from __future__ import annotations

from fastapi import APIRouter, HTTPException
from pathlib import Path
import datetime
from code.api.api import load_llm
from code.api.db import get_conn

router = APIRouter(prefix="/api/draft", tags=["draft"])

# ------------------------------------------------------------
# 1. Assemble all memoir context layers
# ------------------------------------------------------------

def assemble_memoir_context(session_id: str) -> dict:
    conn = get_conn()
    conn.row_factory = lambda cursor, row: {col[0]: row[idx] for idx, col in enumerate(cursor.description)}

    # 1) Person + Profile
    person_row = conn.execute("""
        SELECT p.person_id, pr.field, pr.value
        FROM sessions s
        JOIN people p ON s.person_id = p.person_id
        LEFT JOIN profiles pr ON p.person_id = pr.person_id
        WHERE s.session_id = ?
    """, (session_id,)).fetchall()

    if not person_row:
        raise HTTPException(404, "No profile found for this session")

    person_id = person_row[0]["person_id"]
    profile_json = {row["field"]: row["value"] for row in person_row if row["field"]}

    # 2) Timeline
    timeline_rows = conn.execute("""
        SELECT event_date, event_title, event_description, location_name
        FROM timeline_events
        WHERE person_id = ?
        ORDER BY event_date ASC
    """, (person_id,)).fetchall()

    timeline_txt = "\n".join([
        f"- {r['event_date']}: {r['event_title']} ({r['location_name']}) - {r['event_description']}"
        for r in timeline_rows
    ]) or "No timeline events recorded."

    # 3) Media
    media_rows = conn.execute("""
        SELECT description, taken_at, location_name
        FROM media
        WHERE person_id = ?
    """, (person_id,)).fetchall()

    media_txt = "\n".join([
        f"- Image from {r['taken_at']} at {r['location_name']}: {r['description']}"
        for r in media_rows
    ]) or "No media descriptions available."

    # 4) Transcript
    turn_rows = conn.execute("""
        SELECT role, content
        FROM chat_turns
        WHERE session_id = ?
        ORDER BY id ASC
    """, (session_id,)).fetchall()

    transcript_txt = "\n".join([
        f"{r['role'].upper()}: {r['content']}"
        for r in turn_rows
    ])

    conn.close()

    return {
        "profile_json": profile_json,
        "timeline_events": timeline_txt,
        "media_descriptions": media_txt,
        "chat_turns": transcript_txt,
    }


# ------------------------------------------------------------
# 2. Load Master Prompt Template
# ------------------------------------------------------------

MASTER_PROMPT = """
### ROLE
You are a master non-fiction editor and co-writer. Your style blends the observational wit of Mark Twain, the narrative precision of Joan Didion, and the historical depth of Doris Kearns Goodwin.

### OBJECTIVE
Write a cohesive, first-person memoir chapter based on the provided local data. The goal is to produce a narrative that is as readable as a novel and as accurate as a history book.

### DATA SOURCES
1. PROFILE (Hard Facts):
{profile_json}

2. TIMELINE (Chronology):
{timeline_events}

3. MEDIA (Atmosphere):
{media_descriptions}

4. TRANSCRIPT (Voice & Emotion):
{chat_turns}

### INSTRUCTIONS
- Use the speaker's voice from the transcript.
- Anchor events using the timeline dates.
- Integrate media descriptions as sensory details.
- Begin with the earliest timeline event.
- End with a transition to the next phase of life.
- Preserve emotional truth over perfect chronology.

### CHAPTER OUTPUT
Write Chapter 1: Beginnings
"""


# ------------------------------------------------------------
# 3. Drafting Endpoint
# ------------------------------------------------------------

@router.post("/chapter")
async def draft_chapter(session_id: str):
    model, tok = load_llm()

    context = assemble_memoir_context(session_id)
    prompt = MASTER_PROMPT.format(**context)

    inputs = tok(prompt, return_tensors="pt").to(model.device)
    out = model.generate(
        **inputs,
        max_new_tokens=4096,
        temperature=0.7,
        top_p=0.9,
    )

    chapter = tok.decode(out[0][inputs["input_ids"].shape[1]:], skip_special_tokens=True)

    # Save to disk
    export_dir = Path("/home/chris/lorevox_data/exports/memoir_drafts")
    export_dir.mkdir(parents=True, exist_ok=True)

    filename = export_dir / f"chapter1_{session_id}_{datetime.datetime.now().strftime('%Y%m%d_%H%M%S')}.md"
    filename.write_text(chapter, encoding="utf-8")

    return {
        "ok": True,
        "chapter": chapter,
        "saved_to": str(filename),
    }
